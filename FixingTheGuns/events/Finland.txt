				}

		}

		name = finland.3.b

				}
			}
		}

		trigger = {

				has_war = no

					has_government = communism
				}

			limit = {
				has_equipment = {
					infantry_equipment > 500
				}
			}

				target = FIN
				type = infantry_equipment

			}
		}

				has_equipment = {
					artillery_equipment > 250
				}
			}
			send_equipment = {
				target = FIN
				type = artillery_equipment
				amount = 250
			}
		}
		add_ai_strategy = {
			type = support
			id = FIN
			value = 100
		}
		FIN = {
			country_event = { id = finland.5 days = 3 }
		}
	}

	option = {
		name = finland.3.c

			factor = 1

				factor = 0
				is_historical_focus_on = yes
			}

					tag = SOV
					ratio > 0.2
				}

					has_government = democratic
					strength_ratio = {
						tag = SOV 
						ratio > 0.3
					}
				}
			}			
		}

		declare_war_on = {
			target = SOV
			type = take_state
			generator = { 195 208 } # Leningrand and Luga

			country_event = { id = finland.6 days = 3 }

				white_peace = {
